
# Sensorgnome boot configuration v1 for Raspberry PIs

# CM3+, e.g. SensorStation V1
[board-type=10]
dtparam=i2c_arm=on
dtparam=i2c1=on
dtoverlay=i2c-rtc,ds3231 # RTC
dtoverlay=pps-gpio,gpiopin=13 # GPS PPS input pin
dtoverlay=pi3-disable-bt
gpio=24=op,dh # drive high: power USB hubs
gpio=25=op,dl # drive low: disable modem power
gpio=19=op,dh # drive high: power ADC
core_freq=250 # required to use uart1
dtparam=spi=on # screen uses SPI
# don't enable the modem tx/rx 'cause it's a useless 2G device
#dtoverlay=uart1,txd1_pin=32,rxd1_pin=33 # modem tx/rx

[all]
enable_uart=1
dtparam=act_led_trigger=heartbeat
